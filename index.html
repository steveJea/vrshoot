// 타이틀 화면에서 아무 키나 / 클릭해도 시작되도록 강화 function startFromTitle() { if (gameState === 'title') { showScreen('select-screen'); renderSelect(); } } document.addEventListener('keydown', function(e) { startFromTitle(); // 치트 코드 (셀렉트 화면에서만) if (gameState === 'select') { if (e.key.length === 1 && /[a-zA-Z]/.test(e.key)) { cheatBuffer += e.key.toLowerCase(); if (cheatBuffer.length > 25) cheatBuffer = cheatBuffer.slice(-25); for (const code in CHEATS) { if (cheatBuffer.endsWith(code)) { const boss = bosses[CHEATS[code]]; addBossCard(boss); cheatBuffer = ""; break; } } } } }); // 마우스 클릭으로도 시작 가능하게 document.getElementById('title-screen').addEventListener('click', startFromTitle);
